Algorithms

Dijkstra's Algorithm (For Shortest Path Calculations)

Objective: Find the shortest path from one node to another in the network, ensuring fast access to key locations like hospitals in the event of a disaster.

Application: This algorithm can help determine the quickest routes between transport hubs, schools, and hospitals, ensuring that disaster response teams can reach critical areas quickly.

Prim's Algorithm (For Minimum Spanning Tree)

Objective: Ensure all the nodes in the network are connected in the most cost-efficient way, helping to reduce the overall risk of disaster response inefficiency.

Application: Prim’s algorithm can be used to design a robust road network that connects all important nodes (e.g., transport hubs, hospitals, and schools) at the minimum cost, ensuring accessibility even during emergencies.

Kruskal's Algorithm (For Minimum Spanning Tree)

Objective: Similar to Prim's algorithm, Kruskal's algorithm can also help connect all important nodes while minimizing the total connection cost.

Application: Kruskal's algorithm can be used to build a disaster response network that optimizes road connectivity between transport hubs, schools, and hospitals, ensuring that resources and teams can be deployed efficiently during emergencies.

Floyd-Warshall Algorithm (For All-Pairs Shortest Paths)

Objective: Calculate the shortest paths between every pair of nodes.

Application: This algorithm is useful if the platform needs to provide shortest-path information between all possible combinations of nodes (e.g., from any transport hub to any hospital or school).

#include <iostream>
#include <vector>
#include <climits>
#include <queue>

using namespace std;

class Graph {
public:
    vector<vector<int>> adjMatrix; 
    vector<string> nodes; 

    
    Graph() {
        nodes = {"T1", "T2", "T3", "S1", "S2", "S3", "S4", "H1", "H2"};
        adjMatrix = {
            {0, 12, 18, 10, 15, 22, 25, 8, 20},
            {12, 0, 10, 15, 18, 20, 28, 22, 12},
            {18, 10, 0, 20, 25, 15, 12, 25, 18},
            {10, 15, 20, 0, 8, 12, 18, 10, 22},
            {15, 18, 25, 8, 0, 10, 15, 12, 20},
            {22, 20, 15, 12, 10, 0, 5, 15, 18},
            {25, 28, 12, 18, 15, 5, 0, 20, 10},
            {8, 22, 25, 10, 12, 15, 20, 0, 8},
            {20, 12, 18, 22, 20, 18, 10, 8, 0}
        };
    }

    
    void dijkstra(int start) {
        int n = nodes.size();
        vector<int> dist(n, INT_MAX); 
        vector<bool> visited(n, false); 

        dist[start] = 0;

       
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        pq.push({0, start}); // Push the starting node with distance 0

        while (!pq.empty()) {
            int u = pq.top().second; // Get the node with the smallest distance
            pq.pop();

           
            if (visited[u]) continue;
            visited[u] = true;

           
            for (int v = 0; v < n; ++v) {
                if (!visited[v] && adjMatrix[u][v] != 0) {
                    int newDist = dist[u] + adjMatrix[u][v];
                    if (newDist < dist[v]) {
                        dist[v] = newDist;
                        pq.push({dist[v], v});
                    }
                }
            }
        }

       
        cout << "Shortest paths from " << nodes[start] << ":\n";
        for (int i = 0; i < n; ++i) {
            if (dist[i] == INT_MAX) {
                cout << nodes[i] << " is unreachable.\n";
            } else {
                cout << "To " << nodes[i] << ": " << dist[i] << " km\n";
            }
        }
    }
};

int main() {
   
    Graph g;

   
    string startNode;
    cout << "Enter the starting node (e.g., T1, T2, S1, H1, etc.): ";
    cin >> startNode;

   
    vector<string> nodeList = {"T1", "T2", "T3", "S1", "S2", "S3", "S4", "H1", "H2"};
    int startIndex = -1;

  
    for (int i = 0; i < nodeList.size(); ++i) {
        if (nodeList[i] == startNode) {
            startIndex = i;
            break;
        }
    }

  
    if (startIndex != -1) {
        
        g.dijkstra(startIndex);
    } else {
        cout << "Invalid node. Please enter a valid node from the graph.\n";
    }

    return 0;
}
